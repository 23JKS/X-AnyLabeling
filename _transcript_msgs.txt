核心的 numpy 拟合函数（直接从 `ArcConsistencyLoss` 移植）：

```python
def fit_arc(ys, xs, weights, origin):
    """加权 LLS 拟合过原点圆: rho = a·cosθ + b·sinθ。
    返回 (a, b, R, phi, nrmse)；失败返回 None。"""
    ox, oy = origin
    dx = xs.astype(np.float32) - ox
    dy = ys.astype(np.float32) - oy
    rho = np.sqrt(dx * dx + dy * dy)
    keep = rho > 5.0                      # RHO_MIN：太靠近原点 θ 不稳定
    if keep.sum() < 10:
        return None
    rho, w = rho[keep], weights[keep]
    theta = np.arctan2(dy[keep], dx[keep])
    c, s = np.cos(theta), np.sin(theta)

    w_sum = w.sum()
    ridge = 1e-4 * w_sum
    M = np.array([[(w*c*c).sum() + ridge, (w*c*s).sum()],
                  [(w*c*s).sum(), (w*s*s).sum() + ridge]])
    v = np.array([(w*rho*c).sum(), (w*rho*s).sum()])
    a, b = np.linalg.solve(M, v)
    R = 0.5 * np.hypot(a, b)
    if R < 1.0:
        return None
    resid = rho - a*c - b*s
    nrmse = np.sqrt(max((w*resid*resid).sum() / w_sum, 1e-8)) / R
    phi = np.arctan2(b, a)
    return a, b, R, phi, nrmse
```

后处理逻辑（每块板子聚类之后执行，开关开启且检出原点时）：

```python
# 阈值参考：合成测试真弧 nrmse≈0.005，建议初始值
SPLIT_NRMSE = 0.05      # 残差超过 → 疑似误合并，尝试切分
MERGE_NRMSE = 0.03      # 两簇各自残差都小
MERGE_R_TOL = 0.10      # |R1-R2|/max(R) < 10%
MERGE_PHI_TOL = 0.15    # |φ1-φ2| < 0.15 rad
MERGE_GAP = 40          # 端点最近距离 < 40 px

def _arc_refine(self, instance_map, sem_prob, embedding, origin, pp):
    # --- 1. 误合并切分 ---
    for inst_id in [i for i in np.unique(instance_map) if i > 0]:
        ys, xs = np.where(instance_map == inst_id)
        fit = fit_arc(ys, xs, sem_prob[ys, xs], origin)
        if fit is None or fit[4] <= SPLIT_NRMSE:
            continue
        # 残差过大 → 用更小 eps 对该簇单独重聚类
        sub_emb = embedding[:, ys, xs].T
        sub_emb /= np.clip(np.linalg.norm(sub_emb, axis=1, keepdims=True), 1e-8, None)
        sub_labels = DBSCAN(eps=pp["dbscan_eps"] * 0.6,
                            min_samples=pp["dbscan_min_samples"]).fit_predict(sub_emb)
        if len(set(sub_labels) - {-1}) >= 2:   # 确实分得开才替换
            instance_map[ys, xs] = 0
            for k, lbl in enumerate(sorted(set(sub_labels) - {-1})):
                sel = sub_labels == lbl
                instance_map[ys[sel], xs[sel]] = instance_map.max() + 1

    # --- 2. 断裂合并 ---
    fits = {}
    for inst_id in [i for i in np.unique(instance_map) if i > 0]:
        ys, xs = np.where(instance_map == inst_id)
        fit = fit_arc(ys, xs, sem_prob[ys, xs], origin)
        if fit is not None and fit[4] < MERGE_NRMSE:
            fits[inst_id] = (fit, ys, xs)
    ids = list(fits)
    for i, id1 in enumerate(ids):
        for id2 in ids[i+1:]:
            (_, _, R1, phi1, _), ys1, xs1 = fits[id1]
            (_, _, R2, phi2, _), ys2, xs2 = fits[id2]
            if abs(R1 - R2) / max(R1, R2) > MERGE_R_TOL:
                continue
            dphi = abs(phi1 - phi2)
            if min(dphi, 2*np.pi - dphi) > MERGE_PHI_TOL:
                continue
            # 端点最近距离（用少量边界点近似即可）
            if _min_endpoint_gap(ys1, xs1, ys2, xs2) < MERGE_GAP:
                instance_map[instance_map == id2] = id1
    return instance_map
```
